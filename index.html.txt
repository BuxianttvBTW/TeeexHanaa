    <script>
        // Get the button element by ID
        const button = document.getElementById("btn");

        // Get the audio element by ID
        const music = document.getElementById("bg-music");

        // Array of romantic messages to display
        const notes = [
            "Iâ€™m so lucky to know you.",
            "Every moment is more beautiful with you.",
            "When you smile, the world shines brighter.",
            "Everything is more meaningful with you.",
            "My heart beats only for you.",
            "You are my most precious one.",
            "My life became complete with you.",
            "Being with you is the best feeling.",
            "Everything is more fun with you.",
            "Even thinking of you makes me happy."
        ];

        // Handle button click event
        button.addEventListener("click", () => {
            button.classList.add("expand");
            // Add expand animation class to the button

            setTimeout(() => {
                button.style.display = "none";
                // Hide the button after animation

                music.play();
                // Start playing background music

                showNotes();
                // Display romantic notes
            }, 3000);
            // Wait 3 seconds before executing
        });

        // Function to display notes at random positions
        function showNotes() {
            const width = window.innerWidth;
            // Get screen width

            const height = window.innerHeight;
            // Get screen height

            const usedPositions = [];
            // Store used positions to prevent overlap

            notes.forEach((note, index) => {
                const div = document.createElement("div");
                // Create a new div element

                div.classList.add("notes");
                // Assign notes CSS class

                div.innerText = note;
                // Set the note text

                document.body.appendChild(div);
                // Add the note to the page

                requestAnimationFrame(() => {
                    const rect = div.getBoundingClientRect();
                    // Get element size after rendering

                    let x, y, tries = 0;
                    // Variables for positioning

                    do {
                        x = Math.random() * (width - rect.width);
                        // Random horizontal position

                        y = Math.random() * (height - rect.height);
                        // Random vertical position

                        tries++;
                        // Increase attempt counter
                    } while (
                        isOverlapping(x, y, rect.width, rect.height, usedPositions)
                        && tries < 100
                    );
                    // Retry if overlapping with previous notes

                    div.style.left = `${x}px`;
                    // Apply horizontal position

                    div.style.top = `${y}px`;
                    // Apply vertical position

                    usedPositions.push({
                        x,
                        y,
                        width: rect.width,
                        height: rect.height
                    });
                    // Save used position

                    setTimeout(() => div.classList.add("show"), index * 2500);
                    // Show notes one by one with delay
                });
            });
        }

        // Function to check if a note overlaps with existing notes
        function isOverlapping(x, y, width, height, positions) {
            return positions.some(pos =>
                !(
                    x + width < pos.x ||
                    x > pos.x + pos.width ||
                    y + height < pos.y ||
                    y > pos.y + pos.height
                )
            );
            // Return true if any overlap is detected
        }
